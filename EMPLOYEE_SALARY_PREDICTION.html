w0=y_mean-w1*x_mean¶

w1= sum(x-x_mean)*(y-y_mean)/sum(x-x_mean)^2¶

In [27]:
# Mean_X and Mean_Y
mean_x=np.mean(x)
mean_y=np.mean(y)

# Total length of the Dataset
n=len(x)

# Using The formula caluculate the w0 and w1
numer=0
denom=0

for i in range(n):
    numer+=(x[i]-mean_x)*(y[i]-mean_y) 
    denom+=(x[i]-mean_x)**2
    
w1=numer/denom
w0=mean_y-(w1*mean_x)
print('The Coefficients are',w0,w1)
